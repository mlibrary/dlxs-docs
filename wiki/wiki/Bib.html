<!-- each bibliographic database consists of entries -->

<!ELEMENT BIBDB (GROUP)+>
<!ELEMENT GROUP (A+)>

<!-- record element, called "A", is based on TEI "biblFull" -->

<!-- A = RECORD -->
<!ELEMENT A 		(B, C?, D*, E?, F?, G*, H*, I2?, J*, FMT?, LANG*, TYPE*, NORM*, INST?, GENR*, ABST*, REL*)>
<!-- B = TITLESTMT -->
<!ELEMENT B 		((K)*,(L|M|N|O|Q|R)*)>
<!-- C = EDITIONSTMT -->
<!ELEMENT C 		((S,(R)*)|(P)+)>
<!-- D = EXTENT -->
<!ELEMENT D 		(#PCDATA)>
<!-- E = PUBLICATIONSTMT -->
<!ELEMENT E 		(((T?, U?) | AG), V?, W*, X?, YR*)+>
<!-- F = SERIESSTMT -->
<!ELEMENT F 		(K+, Z?)> <!-- added cite (i.e., Z) for serial ref, jpw -->
<!-- G = NOTESSTMT -->
<!ELEMENT G 		((AA)+)>
<!-- H = SOURCEDESC -->
<!ELEMENT H 		(P|AB)+>
<!-- I2 = TEXTCLASS -->
<!ELEMENT I2 	    (SG*,KW*)>
<!-- J = HLDGSTMT -->
<!ELEMENT J 		(LOC*,URL*,COLLS?)>
<!-- K = TITLE -->
<!ELEMENT K	        (#PCDATA|I|GRK|SUB|SUP|BR)*>
<!-- L = AUTHOR -->
<!ELEMENT L	        (#PCDATA|BR)*>
<!-- M = EDITOR -->
<!ELEMENT M	        (#PCDATA)>
<!-- N = SPONSOR -->
<!ELEMENT N	        (#PCDATA)>
<!-- O = FUNDER -->
<!ELEMENT O	        (#PCDATA)>
<!-- P = P -->
<!ELEMENT P	        (#PCDATA|I|SUP)*>
<!-- Q = PRINCIPAL -->
<!ELEMENT Q	        (#PCDATA)>
<!-- R = RESPSTMT -->
<!ELEMENT R 		(AD,AE)>
<!-- S = EDITION -->
<!ELEMENT S	        (#PCDATA|I)*>
<!-- T = PUBLISHER -->
<!ELEMENT T	        (#PCDATA|I)*>
<!-- U = PUBPLACE -->
<!ELEMENT U	        (#PCDATA)>
<!-- V = ADDRESS -->
<!ELEMENT V        (#PCDATA|VV)*>
<!-- VV = STREET AND CONTACT INFO FOR ETHNIC DIRECTORY -->
<!ELEMENT VV       (ST*, PH*, FX*, EM*, URL*)>
<!-- ST = STREET -->
<!ELEMENT ST       (#PCDATA)>
<!-- PH = PHONE -->
<!ELEMENT PH       (#PCDATA)>
<!-- FX = FAX -->
<!ELEMENT FX       (#PCDATA)>
<!-- EM = EMAIL -->
<!ELEMENT EM       (#PCDATA)>
<!-- W = IDNO -->
<!ELEMENT W	       (#PCDATA|SUP)*>
<!-- X = AVAILABILITY -->
<!ELEMENT X 	   (#PCDATA|P|BR)*>
<!-- Z = CITE -->
<!ELEMENT Z 	   ((P)|(YR?, VO?, IS?, PG?))>
<!-- AA = NOTE -->
<!ELEMENT AA 	   (#PCDATA|P|BR|I|SUB|SUP)*>
<!-- AB = BIBL -->
<!ELEMENT AB       (#PCDATA|K|L|S|N|M|AH|T|U|P|B|D|E|F|C|G|I2)*>
<!-- AD = NAME -->
<!ELEMENT AD 	   (#PCDATA)>
<!-- AE = RESP -->
<!ELEMENT AE 	   (#PCDATA)>
<!-- AF = TERM -->
<!ELEMENT AF 	   (#PCDATA|I)*>
<!-- AG = IMPRINT -->
<!ELEMENT AG       (#PCDATA)>
<!-- AH = BIBLSCOPE -->
<!ELEMENT AH       (#PCDATA|I)*>
<!-- BR - HTML BR -->
<!ELEMENT BR  (#PCDATA)*>
<!-- IS = ISSUE -->
<!ELEMENT IS 	   (#PCDATA)>
<!-- I = ITALIC -->
<!ELEMENT I	       (#PCDATA)>
<!-- KW = KEYWORD -->
<!ELEMENT KW       (AF)+>
<!-- PG = PAGE -->
<!ELEMENT PG 	   (#PCDATA)>
<!-- SG = SUBGRP -->
<!ELEMENT SG 	   (SU)*> <!-- SU/SG from BAS; keep and map to keyword/term? -->
<!-- SU = SUBJECT from BAS -->
<!ELEMENT SU 	   (#PCDATA|BR)*> 
<!-- SUB = SUBSCRIPT -->
<!ELEMENT SUB 	   (#PCDATA)> 
<!-- SUP = SUPERSCRIPT -->
<!ELEMENT SUP 	   (#PCDATA)> 
<!-- VO = VOLUME -->
<!ELEMENT VO 	   (#PCDATA)>
<!-- YR = YEAR -->
<!ELEMENT YR 	   (#PCDATA)>
<!-- LOC = LOCATION -->
<!ELEMENT LOC      (#PCDATA|I)*>
<!-- URL = URL -->
<!ELEMENT URL      (#PCDATA|I)*>
<!-- GRK = GREEK -->
<!ELEMENT GRK      (#PCDATA|I)*>
<!-- COLLS = COLLECTIONS -->
<!ELEMENT COLLS    (COLL)+>
<!ELEMENT COLL     (#PCDATA) > 
<!-- FMT = format, e.g., HTML, MPEG, JPEG, WAV -->
<!ELEMENT FMT      (#PCDATA)>
<!-- LANG = languages of the digital object -->
<!ELEMENT LANG     (#PCDATA)>
<!-- TYPE = type, e.g., image, audio, book, article --> 
<!ELEMENT TYPE     (#PCDATA)>
<!-- NORM = normalized type values -->   
<!ELEMENT NORM     (#PCDATA)> 
<!-- INST = institution -->  
<!ELEMENT INST     (#PCDATA)> 
<!-- GENR  -->
<!ELEMENT GENR     (#PCDATA)>
<!-- ABST  -->
<!ELEMENT ABST     (#PCDATA)>
<!-- REL  = reference to related resource -->
<!ELEMENT REL 	   (#PCDATA|URL)*>
